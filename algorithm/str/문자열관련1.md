/** 문자열이 주어졌을때 중복된 글자가 있는지 확인하기 */
var str = "abcdefghia";
var hashMap = {};
for(var i=0; i<str.length; i++){
    if(hashMap[str[i]]){
        console.log("중복 ", str[i]);
        break;
    }
    else{
        hashMap[str[i]] = str[i];
    }
}

console.log(strUrl.replace(/ /gi, "%20"));
