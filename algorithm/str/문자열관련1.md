/** 문자열이 주어졌을때 중복된 글자가 있는지 확인하기 */
var str = "abcdefghia";
var hashMap = {};
for(var i=0; i<str.length; i++){
    if(hashMap[str[i]]){
        console.log("중복 ", str[i]);
        break;
    }
    else{
        hashMap[str[i]] = str[i];
    }
}

/** 문자열 편집방법중 문자삽입,문자삭제,문자교체가 있을때 편집 횟수가 1회 이내인지를 확인*/
var str1 = "pale";
var str2 = "bake";
var hashMap = {};
console.log(str1, str2);

str1.split("").forEach(function(c){
    if(hashMap[c]){
        hashMap[c]++;
    }    
    else{
        hashMap[c] = 1;
    }
})

var hashMap2 = {};
str2.split("").forEach(function(c){
    if(hashMap[c]){
        hashMap[c]--;
        if(hashMap[c] == 0){
            delete hashMap[c];
        }
    }
    else{
        if(hashMap2[c]){
            hashMap2[c]++;
        }
        else{
            hashMap2[c] = 1;
        }
    }
})

var checkSum = 0;
Object.keys(hashMap).forEach(function(obj){
    checkSum += hashMap[obj];
});
